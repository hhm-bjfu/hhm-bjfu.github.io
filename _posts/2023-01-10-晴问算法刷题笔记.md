#### [晴问算法——vector的常见用法详解](https://sunnywhy.com/sfbj/6/1)

> vector有通过下标访问和迭代器访问两种方式。

```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
    vector<int> v;
    int n; cin >> n;
    for (int i = 0; i < n; i ++) {
        int tmp; cin >> tmp;
        v.push_back(tmp);
    }

    // 通过下标访问
    // for (int i = 0; i < n; i ++) {
    //     cout << v[i];
    //     if (i != n - 1) cout << " ";
    // }
    // cout << endl;

    // 通过迭代器访问
    for (vector<int>::iterator it = v.begin(); it != v.end(); it ++) {
        cout << *it;
        if (it != v.end() - 1) cout << " ";
    }
    cout << endl;

    return 0;
}
```

```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
```
```cpp
    int n; cin >> n;
    vector<int> v;
    for (int i = 0; i < n; i ++) {
        int tmp; cin >> tmp;
        v.push_back(tmp);
    }

    int x, k1, k2;
    cin >> x >> k1 >> k2;

    // 插入元素，插入后元素x占据插入位置
    v.insert(v.begin() + k1, x);
    
    // 删除元素，有针对迭代器单元素删除和区间删除，这里使用前者
    v.erase(v.begin() + k2);

    for (int i = 0; i < v.size(); i ++) {
        cout << v[i];
        if (i != v.size() - 1)
            cout << " ";
    }

    return 0;
}
```

```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
```
```cpp
    int n; cin >> n;
    vector<int> vs[n];

    for (int i = 0; i < n; i ++) {
        int k; cin >> k;
        while (k --) {
            int tmp; cin >> tmp;
            vs[i].push_back(tmp);
        }
    }

    for (int i = 0; i < n; i ++) {
        for (int j = 0; j < vs[i].size(); j ++) {
            cout << vs[i][j];
            if (j != vs[i].size() - 1) 
                cout << " ";
        }
        cout << endl;
    }

    return 0;
}
```

#### [晴问算法——set的常见用法详解](https://sunnywhy.com/sfbj/6/2)

```cpp
#include <iostream>
#include <set>

using namespace std;

int main() {
    set<int> s;
    int n; cin >> n;
    int tmp;
    for (int i = 0; i < n; i ++) {
        cin >> tmp;
        s.insert(tmp);
    }

    for (set<int>::iterator it = s.begin(); it != s.end(); it ++) {
        if (it != s.begin()) 
            cout << " ";
        cout << *it;
    }

    return 0;
}
```
