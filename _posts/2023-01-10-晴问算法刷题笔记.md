#### [晴问算法——vector的常见用法详解](https://sunnywhy.com/sfbj/6/1)

> vector有通过下标访问和迭代器访问两种方式。

```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
    vector<int> v;
    int n; cin >> n;
    for (int i = 0; i < n; i ++) {
        int tmp; cin >> tmp;
        v.push_back(tmp);
    }

    // 通过下标访问
    // for (int i = 0; i < n; i ++) {
    //     cout << v[i];
    //     if (i != n - 1) cout << " ";
    // }
    // cout << endl;

    // 通过迭代器访问
    for (vector<int>::iterator it = v.begin(); it != v.end(); it ++) {
        cout << *it;
        if (it != v.end() - 1) cout << " ";
    }
    cout << endl;

    return 0;
}
```

```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> v;
    for (int i = 0; i < n; i ++)
        v.push_back(k);
    for (vector<int>::iterator it = v.begin(); it != v.end(); it ++) {
        cout << *it;
        if (it != v.end() - 1) cout << " ";
    }

    return 0;
}
```

```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> v;
    for (int i = 0; i < n; i ++)
        v.push_back(k);
    for (int i = 0; i < k; i ++)
        v.pop_back();
    cout << v.size() << endl;

    return 0;
}
```
